进一步加强

上述代码还是不能让我们很好的理解base64的过程，因此可以进一步扩展去了解下base64，然后自己写一个本地的base64模块。

base64详解

local_base64.py

# -*- coding: UTF-8 -*-

import string

base64_charset = string.ascii_uppercase + string.ascii_lowercase + string.digits + '+/'

def encode(string):

    base64_bin = ['{:0>8}'.format(str(bin(ord(b))).replace('0b', '')) for b in string]

    result = ''

    nums = len(base64_bin) // 3

    remain = len(base64_bin) % 3

    integral_part = base64_bin[0:3 * nums]

    while integral_part:

        tmp_part = ''.join(integral_part[0:3])

        tmp_part = [int(tmp_part[x: x + 6], 2) for x in [0, 6, 12, 18]]

        result += ''.join([base64_charset[i] for i in tmp_part])

        integral_part = integral_part[3:]

    if remain:

        remain_part = ''.join(base64_bin[3 * nums:]) + (3 - remain) * '0' * 8

        tmp_part = [int(remain_part[x: x + 6], 2) for x in [0, 6, 12, 18]][:remain + 1]

        result += ''.join([base64_charset[i] for i in tmp_part]) + (3 - remain) * '='

    return result

def decode(string):

    if not valid_base64_str(string):

        return ''

    base64_bin = ['{:0>6}'.format(str(bin(base64_charset.index(s))).replace('0b', '')) for s in string if s != '=']

    result = ''

    nums = len(base64_bin) // 4

    remain = len(base64_bin) % 4

    integral_part = base64_bin[0:4 * nums]

    while integral_part:

        tmp_unit = ''.join(integral_part[0:4])

        tmp_unit = [int(tmp_unit[x: x + 8], 2) for x in [0, 8, 16]]

        result += ''.join([chr(i) for i in tmp_unit])

        integral_part = integral_part[4:]

    if remain:

        remain_part = ''.join(base64_bin[nums * 4:])

        tmp_unit = [int(remain_part[i * 8:(i + 1) * 8], 2) for i in range(remain - 1)]

        result += ''.join([chr(i) for i in tmp_unit])

    return result

def valid_base64_str(b_str):

    if len(b_str) % 4:

        return False

    for m in b_str:

        if m not in base64_charset + '=':

            return False

    return True

if __name__ == '__main__':

    s = 'ABC'

    s_encode = encode(s)

    s_decode = decode(s_encode)

    print s_encode

    print

    print s_decode

Challenge 2

挑战2 目的是让你编写一个函数，使得当你输入

1c0111001f010100061a024b53535009181c

让上述字符串，与下列字符串进行异或（XOR）运算

686974207468652062756c6c277320657965

得到的结果应该为

746865206b696420646f6e277420706c6179

只需要进行一个异或运算，感觉起来应该不难，让我们先来了解下XOR运算操作

逻辑运算详解

通过上述运算原理的介绍，可以看出这道题目就是等长度异或，实现起来很容易

exp

# -*- coding: UTF-8 -*-

string_hex = '1c0111001f010100061a024b53535009181c'

string = int(string_hex,16)

string_xor = '686974207468652062756c6c277320657965'

string_xor = int(string_xor,16)

string_result = string ^ string_xor

string_result = hex(string_result)

print 'result :' + string_result[2:-1]

1b37373331363f78151b7f2b783431333d78397828372d363c78373e783a393b3736

既然是编写函数，那必然要了解python中的def

python中的def

先不考虑评分标准，要先找到那个单字符
回想一下ascii码中一共有256个，虽然一个一个尝试对于人来说比较缓慢，但对于计算机来说应该不到一秒钟的事。所以这里决定来进行遍历尝试，也就是俗称的“暴力破解”

暴力破解模块

def brute(string_xor):

    score_pre = 0

    for i in range(128):
        result = ''
        for j in re.findall('.{2}',string_xor):
            k = int(j,16)
            r = chr(i ^ k)
            result += r

这样的话，256种结果都会被存储到result中，接下来解决打分的问题，只需要对每个result进行个测评，然后比较下输出最高的就好了。

exp

# -*- coding: UTF-8 -*-

import re

string_xor = '1b37373331363f78151b7f2b783431333d78397828372d363c78373e783a393b3736'

def judge(s):

    score_tmp = 0

    freqs = {'a':0.0651738,'b':0.0124248,'c':0.0217339,'d':0.0349835,
         'e':0.1041442,'f':0.0197881,'g':0.0158610,'h':0.0492888,
         'i':0.0558094,'j':0.0009033,'k':0.0050529,'l':0.0331490,
         'm':0.0202124,'n':0.0564513,'o':0.0596302,'p':0.0137645,
         'q':0.0008606,'r':0.0497563,'s':0.0515760,'t':0.0729357,
         'u':0.0225134,'v':0.0082903,'w':0.0171272,'x':0.0013692,
         'y':0.0145984,'z':0.0007836,' ':0.1918182
         }
    for x in s.lower():
        if x in freqs:
            score_tmp += freqs[x]

    return score_tmp

def brute(string_xor):

    score_pre = 0

    for i in range(128):
        result = ''
        for j in re.findall('.{2}',string_xor):
            k = int(j,16)
            r = chr(i ^ k)
            result += r
        score = judge(result)

        if score > score_pre:
            score_pre = score
            final = result
            key = chr(i)

    return final,key

final,key = brute(string_xor)

print final
